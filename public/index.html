Supabase GitHub

Go to supabase.com, create a free project, go to Project Settings → API and paste your keys.

create table athletes (id uuid primary key default gen_random_uuid(), name text, program text, due_date text, notes text, meets jsonb default '[]', start_date text, end_date text, churn_reason text, is_active boolean default true, birthday text, birthday_on_phone boolean default false, created_at timestamptz default now());

If you already have the athletes table, add the birthday columns:

alter table athletes add column if not exists birthday text; alter table athletes add column if not exists birthday_on_phone boolean default false;

Also run this for PR tracking:

create table pr_logs (id uuid default gen_random_uuid() primary key, athlete_id uuid references athletes(id) on delete cascade, block text not null, date date not null, squat_pr boolean, bench_pr boolean, dead_pr boolean, squat numeric, bench numeric, dead numeric, notes text, created_at timestamptz default now());

And this for app settings (required for GitHub Pages hosting):

create table if not exists app_settings (key text primary key, value jsonb); alter table app_settings enable row level security; create policy "service role full access" on app_settings using (true) with check (true);
